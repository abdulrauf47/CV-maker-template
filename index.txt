const stepsList = document.getElementById("stepsList").children;

function updateContent(index) {
  // Update active step
  [...stepsList].forEach(li => li.classList.remove("active"));
  stepsList[index].classList.add("active");

  // Update box 1
  const box1 = document.getElementById("toolBox1");
  box1.querySelector("h2").textContent = steps[index][0].title;
  box1.querySelector("p").textContent = steps[index][0].text;
  box1.querySelector("img").src = steps[index][0].img;

  // Update box 2
  const box2 = document.getElementById("toolBox2");
  box2.querySelector("h2").textContent = steps[index][1].title;
  box2.querySelector("p").textContent = steps[index][1].text;
  box2.querySelector("img").src = steps[index][1].img;
}

// Initialize first step
updateContent(0);

// Add click event
[...stepsList].forEach((li, idx) => {
  li.addEventListener("click", () => {
    updateContent(idx);
  });
});
